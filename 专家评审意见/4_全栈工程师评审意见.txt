二、与现有数据模型的关键冲突
2.1 粒度冲突（最严重问题）
维度	原始模型	新需求	冲突说明
时间粒度	月级总量	时/日级分解	原始模型只存储月度值，但要输出时/日级数据 → 需要时间序列分解算法
设备粒度	厂级总量	单台设备级	原始模型没有设备维度 → 需要设备台账+分摊算法
数据性质	实际采集值	模拟生成值（随机偏移）	混合了"实测"和"模拟"两种数据质量 → 需要数据质量标签
结论：新需求实际上是一个数据增强/模拟系统，而非纯粹的采集系统。原始模型是"记录实际数据"，新需求是"基于有限数据生成更细粒度数据"。

决策点A：设备分摊算法（承接之前评审的决策点4）
您流程图中提到两个选项：

路径A（新数据）：依统计最小单元粒度

路径B（上月数据）：单台较上月随机偏移1%以内

技术方案：

python
class DeviceAllocationStrategy(ABC):
    """设备分摊策略抽象类"""
    @abstractmethod
    def allocate(self, total_energy: float, devices: List[Device]) -> Dict[int, float]:
        pass

class HistoricalActualStrategy(DeviceAllocationStrategy):
    """历史实际比例策略（推荐用于路径A）"""
    def allocate(self, total_energy, devices):
        # 基于过去3个月的实际运行比例
        ratios = [d.avg_ratio_last_3_months for d in devices]
        return self._allocate_by_ratio(total_energy, ratios)

class RatedPowerStrategy(DeviceAllocationStrategy):
    """额定功率比例策略（降级方案）"""
    def allocate(self, total_energy, devices):
        ratios = [d.rated_power for d in devices]
        return self._allocate_by_ratio(total_energy, ratios)

class RandomOffsetStrategy(DeviceAllocationStrategy):
    """随机偏移策略（用于路径B）"""
    def __init__(self, offset_range: float = 0.01):  # 1%偏移
        self.offset_range = offset_range
    
    def allocate(self, total_energy, devices):
        # 获取上月的分摊比例
        last_month_ratios = [d.last_month_ratio for d in devices]
        # 添加随机偏移
        current_ratios = [r * (1 + random.uniform(-self.offset_range, self.offset_range)) 
                          for r in last_month_ratios]
        # 归一化
        current_ratios = [r/sum(current_ratios) for r in current_ratios]
        return self._allocate_by_ratio(total_energy, current_ratios)

class TemporalDecompositionStrategy:
    def __init__(self, load_profile_library: LoadProfileLibrary):
        self.profiles = load_profile_library
    
    def decompose(self, monthly_total: float, energy_type: str, season: str) -> List[DailyData]:
        # 1. 获取该能源类型的典型日负荷曲线
        typical_curve = self.profiles.get(energy_type, season)
        
        # 2. 将月度总量按典型曲线分配到每日
        daily_ratios = typical_curve.daily_ratios
        daily_values = [monthly_total * r for r in daily_ratios]
        
        # 3. 如果是路径B（随机偏移），添加±5%波动
        if self.with_random_offset:
            daily_values = self._add_random_offset(daily_values, 0.05)
        
        return daily_values

class LoadProfileLibrary:
    """典型负荷曲线库"""
    def __init__(self):
        # 从历史数据中学习，或使用行业标准曲线
        self.profiles = {
            ("electricity", "summer"): [0.03, 0.03, 0.02, ..., 0.04],  # 31天
            ("electricity", "winter"): [0.04, 0.03, 0.03, ..., 0.03],

决策点C：随机偏移的一致性
需求中出现了两个随机偏移：

重要问题：这两个偏移是否独立？还是需要保证设备级偏移+时间级偏移后的总量回写？

建议方案：

python
# 确保总量守恒
def ensure_total_conservation(allocated_values: List[float], original_total: float) -> List[float]:
    """确保分摊后的总和等于原始总量"""
    current_total = sum(allocated_values)
    if abs(current_total - original_total) > 1e-6:
        # 按比例调整
        scale_factor = original_total / current_total
        return [v * scale_factor for v in allocated_values]
    return allocated_values
