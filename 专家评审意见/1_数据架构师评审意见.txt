针对 4 个关键决策点的评审意见
决策 1：存储方式 — 独立表 daily_energy ✅（强烈推荐）
sql
复制
-- ✅ 推荐设计：独立 daily_energy 表
CREATE TABLE daily_energy (
    id BIGSERIAL PRIMARY KEY,
    date DATE NOT NULL,
    energy_type VARCHAR(20) NOT NULL,  -- electricity, water, diesel, gasoline, gas, steam...
    value DECIMAL(15,2) NOT NULL,
    unit VARCHAR(10),
    source VARCHAR(20) DEFAULT 'simulated',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(date, energy_type)
);

-- 核心索引
CREATE INDEX idx_daily_energy_date ON daily_energy(date DESC);
CREATE INDEX idx_daily_energy_type ON daily_energy(energy_type);
CREATE INDEX idx_daily_energy_date_type ON daily_energy(date, energy_type);
--                          ↑ 复合索引：覆盖最常用查询模式

-- ❌ 不推荐：扩展 production_data
-- 会导致表超宽（不断加列）、NULL 值膨胀、查询需 SELECT 10+ 字段
理由：

独立表是列可扩展的：加能源类型加行不加列
production_data 加列需要 ALTER TABLE，在有锁的情况下危险
SQLite 对宽表的 B-tree 效率低于窄表

-- ✅ 独立偏移：数据更真实，查询时用简单的 UPDATE 即可
UPDATE daily_energy

决策 4：设备分摊 — 按额定功率比例 ✅（初始阶段）
sql
复制
-- ✅ 额定功率比例：简单、可缓存、不需要历史数据
SELECT d.id, d.rated_power / wc.total_power AS ratio
FROM devices d
JOIN (
    SELECT work_center_id, SUM(rated_power) as total_power
    FROM devices GROUP BY work_center_id
) wc ON wc.work_center_id = d.work_center_id;
-- 注意：work_center_id 必须建索引！
紧急索引优化方案（立即执行）
